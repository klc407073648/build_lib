#可执行文件输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${3PARTLIB_EXECUTABLE_OUTPUT_PATH}/jsoncpp)

#头文件目录

#目录源码路径
SET( CONFIG_SRC_DIR ./source )

#代码列表
FILE (GLOB CONFIG_FILES ${CONFIG_SRC_DIR}/*.cpp)

#依赖的LIB
SET(CONFIG_LIB  ${LIB_JSONCPP} ${LIB_STIBEL_COMMON})

#生成 
ADD_EXECUTABLE( globalConfigTest ${CONFIG_FILES})

#连接生成相关库
TARGET_LINK_LIBRARIES( globalConfigTest ${LIB_JSONCPP} ${LIB_YAMLCPP} ${LIB_TINYXML})
