#可执行文件输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${COMLIB_EXECUTABLE_OUTPUT_PATH}/Data/MySQL)

#头文件目录
INCLUDE_DIRECTORIES( ${STIBEL_COMMON_INCLUDE} ${STIBEL_MYSQL_INCLUDE}  ./include )

#目录源码路径
SET( STIBEL_DATA_MYSQL_TEST_SRC_DIR ./source )

#代码列表
FILE (GLOB STIBEL_DATA_MYSQL_TEST_FILES ${STIBEL_DATA_MYSQL_TEST_SRC_DIR}/*.cpp)

#依赖的LIB
SET(STIBEL_DATA_MYSQL_TEST_LIB  ${LIB_STIBEL_MYSQL} ${LIB_STIBEL_COMMON})

#生成 
ADD_EXECUTABLE( MySQLTest ${STIBEL_DATA_MYSQL_TEST_FILES})

#连接生成相关库
TARGET_LINK_LIBRARIES( MySQLTest ${STIBEL_DATA_MYSQL_TEST_LIB} )
