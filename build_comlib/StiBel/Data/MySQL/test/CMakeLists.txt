#可执行文件输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${STIBEL_TEST_OUTPUT_PATH}/Data/MySQL)

#依赖的LIB

#生成 
ADD_EXECUTABLE( DataMySQLTest ${DATA_MYSQL_TEST_FILES})

#连接生成相关库
